# 数据源重构 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把 AI 新闻日报的数据源从「6 源 tab + 综合精选 Top 15」重构为「22 源 + 4 板块 tab + 编辑按语」，去除噪声源、补足覆盖盲区。

**Architecture:** 复用现有 collector / processor 架构。RSS 类源直接扩展 `RSSCollector` 配置；X 改走 RSSHub；国内厂商和政策站走 Firecrawl 抓 SPA；新增 `section_mapper` 把 source 映射成 4 板块；`main.py` 移除综合精选步骤，输出结构 `items + by_source` 改为 `by_section`。

**Tech Stack:** Python 3 + requests + feedparser-style XML 解析（已实现）+ firecrawl-py（新增）+ 现有 DeepSeek API + 静态 Web 前端（vanilla JS）

**Spec:** [docs/superpowers/specs/2026-05-08-数据源重构-design.md](../specs/2026-05-08-数据源重构-design.md)

**测试策略：** 项目无单元测试基础设施（CLAUDE.md 明确"无测试套件、无 lint 配置"）。本计划使用 **smoke test** 替代 TDD：每个 collector 写完后单独跑一次确认能拉到数据，每阶段末跑端到端验证。Smoke 脚本放在 `scripts/smoke/` 目录下，不进 git（加 `.gitignore`）。

```python
"""
通过 RSSHub 抓取 X (Twitter) 白名单用户的推文。
RSSHub 路由：/twitter/user/{username}
输出格式与原 x_collector 兼容：source="X"。
"""
import requests
import xml.etree.ElementTree as ET
from datetime import datetime, timezone
from email.utils import parsedate_to_datetime


class RSSHubXCollector:
    def __init__(self, config: dict):
        self.cfg = config.get("rsshub", {})
        self.endpoint = self.cfg.get("endpoint", "http://localhost:1200")
        self.users = self.cfg.get("x_whitelist", [])
        self.max_per_user = self.cfg.get("max_per_user", 5)
        self.headers = {
            "User-Agent": "Mozilla/5.0 (compatible; AI-News-Bot/1.0)",
        }

    def _parse_date(self, date_str: str) -> str:
        if not date_str:
            return datetime.now(timezone.utc).isoformat()
        try:
            return parsedate_to_datetime(date_str).isoformat()
        except Exception:
            return datetime.now(timezone.utc).isoformat()

    def _fetch_user(self, username: str) -> list:
        url = f"{self.endpoint}/twitter/user/{username}"
        try:
            resp = requests.get(url, headers=self.headers, timeout=20)
            if resp.status_code != 200:
                print(f"[RSSHubX] {username} HTTP {resp.status_code}")
                return []
            root = ET.fromstring(resp.content)
            items = []
            for it in root.findall(".//item"):
                title = (it.findtext("title") or "").strip()
                link = (it.findtext("link") or "").strip()
                desc = (it.findtext("description") or "").strip()
                pub = it.findtext("pubDate")
                if not title:
                    continue
                # 只保留含外链或是 quote tweet 的
                # 简单判定：description 里出现 http(s):// 链接，或 title 包含 RT/QT 标识
                has_link = "http://" in desc or "https://" in desc
                if not has_link:
                    continue
                items.append({
                    "id": f"x_{hash(link or title) & 0xFFFFFFFF:08x}",
                    "title": title[:200],
                    "content": (desc or title)[:1500],
                    "url": link,
                    "source": "X",
                    "created_at": self._parse_date(pub),
                    "likes": 0,  # RSSHub 不带这些计数，热度评分会偏低
                    "retweets": 0,
                    "replies": 0,
                    "author": username,
                    "author_followers": 100000,  # 白名单都是大 V，给个高 follower 让热度评分加成生效
                })
            return items[:self.max_per_user]
        except ET.ParseError as e:
            print(f"[RSSHubX] {username} XML error: {e}")
            return []
        except Exception as e:
            print(f"[RSSHubX] {username} error: {e}")
            return []

    def fetch(self) -> list:
        all_items = []
        for user in self.users:
            batch = self._fetch_user(user)
            all_items.extend(batch)
            print(f"[RSSHubX] @{user}: {len(batch)} items")
        print(f"[RSSHubX] Total {len(all_items)} from {len(self.users)} users")
        return all_items
```

### Task 5.1：创建 firecrawl_collector.py

**Files:**

- [ ] **Step 1：写 collector**

```python
"""
通过 Firecrawl 抓取国内厂商博客 + 政策站点（这些站点是 SPA 或无 RSS）。
策略：每个目标抓首页/列表页 → Firecrawl 返回 markdown → 用启发式规则提取标题列表。
不做深度内容抓取（主流水线后续步骤的 article_fetcher 会再抓一次完整正文）。
"""
import os
from datetime import datetime, timezone
from firecrawl import FirecrawlApp


class FirecrawlCollector:
    def __init__(self, config: dict):
        self.cfg = config.get("firecrawl", {})
        self.api_key = os.getenv("FIRECRAWL_API_KEY", "")
        self.targets = self.cfg.get("targets", [])
        self.max_per_target = self.cfg.get("max_per_target", 5)
        self.app = FirecrawlApp(api_key=self.api_key) if self.api_key else None

    def _scrape(self, url: str) -> str:
        """返回页面 markdown 内容；失败返回空串。"""
        if not self.app:
            return ""
        try:
            result = self.app.scrape_url(url, params={"formats": ["markdown"]})
            return result.get("markdown", "") if isinstance(result, dict) else ""
        except Exception as e:
            print(f"[Firecrawl] scrape {url} error: {e}")
            return ""

    def _extract_links_from_md(self, md: str, base_url: str) -> list:
        """从 markdown 里抽取 [title](url) 形式的链接。
        过滤掉导航栏/footer 类短文本（<10 字符的 anchor text）。"""
        import re
        from urllib.parse import urljoin
        links = []
        seen = set()
        for m in re.finditer(r"\[([^\]]+)\]\((https?://[^\)]+)\)", md):
            title = m.group(1).strip()
            url = m.group(2).strip()
            if len(title) < 10 or url in seen:
                continue
            seen.add(url)
            links.append({"title": title, "url": urljoin(base_url, url)})
        return links

    def _fetch_target(self, target: dict) -> list:
        name = target.get("name", "")
        url = target.get("url", "")
        section = target.get("section", "morning")  # 提示后续板块映射
        if not url or "TBD" in url:
            print(f"[Firecrawl] skip {name}: URL 未配置")
            return []
        md = self._scrape(url)
        if not md:
            return []
        links = self._extract_links_from_md(md, url)
        items = []
        for link in links[:self.max_per_target]:
            items.append({
                "id": f"fc_{hash(link['url']) & 0xFFFFFFFF:08x}",
                "title": link["title"],
                "content": link["title"],  # 列表页只有标题；正文交给后续 article_fetcher
                "url": link["url"],
                "source": name,
                "created_at": datetime.now(timezone.utc).isoformat(),  # 列表页无可靠 pubDate
                "likes": 0, "retweets": 0, "replies": 0,
                "author": name, "author_followers": 0,
                "_section_hint": section,  # 给 section_mapper 用
            })
        return items

    def fetch(self) -> list:
        if not self.app:
            print("[Firecrawl] FIRECRAWL_API_KEY 未配置，跳过")
            return []
        all_items = []
        for target in self.targets:
            batch = self._fetch_target(target)
            all_items.extend(batch)
            print(f"[Firecrawl] {target.get('name')}: {len(batch)} items")
        print(f"[Firecrawl] Total {len(all_items)} items from {len(self.targets)} targets")
        return all_items
```

## Phase 6：新增 HuggingFace Trending Models Collector

### Task 6.1：创建 hf_collector.py

**Files:**
- Create: `src/collectors/hf_collector.py`

- [ ] **Step 1：写 collector**

```python
"""
HuggingFace Trending Models 采集。
HF 没有官方"trending"端点，但 huggingface_hub 库的 list_models(sort="downloads") 可近似。
策略：list_models(sort="downloads", limit=N, filter="text-generation")，把每个模型当成一个条目。
"""
import requests
from datetime import datetime, timezone


class HFCollector:
    def __init__(self, config: dict):
        self.cfg = config.get("huggingface", {})
        self.max_results = self.cfg.get("max_results", 15)
        self.sort_by = self.cfg.get("sort", "trendingScore")  # trendingScore / downloads / likes
        self.endpoint = "https://huggingface.co/api/models"

    def fetch(self) -> list:
        params = {
            "sort": self.sort_by,
            "direction": -1,
            "limit": self.max_results,
            "filter": "text-generation",
        }
        try:
            resp = requests.get(self.endpoint, params=params, timeout=20)
            resp.raise_for_status()
            models = resp.json()
        except Exception as e:
            print(f"[HF] fetch error: {e}")
            return []

        items = []
        for m in models:
            model_id = m.get("modelId") or m.get("id") or ""
            if not model_id:
                continue
            likes = m.get("likes", 0)
            downloads = m.get("downloads", 0)
            updated = m.get("lastModified") or m.get("createdAt")
            tags = ", ".join(m.get("tags", [])[:5])
            items.append({
                "id": f"hf_{hash(model_id) & 0xFFFFFFFF:08x}",
                "title": model_id,
                "content": f"HuggingFace 模型: {model_id}\n下载: {downloads} | 点赞: {likes}\n标签: {tags}",
                "url": f"https://huggingface.co/{model_id}",
                "source": "HuggingFace",
                "created_at": updated or datetime.now(timezone.utc).isoformat(),
                "likes": likes,
                "retweets": 0,
                "replies": 0,
                "author": model_id.split("/")[0] if "/" in model_id else "HuggingFace",
                "author_followers": 0,
                "stars": likes,
                "_section_hint": "weekend",
            })
        print(f"[HF] {len(items)} trending models")
        return items
```

### Task 7.3：新增 section_mapper.py

**Files:**
- Create: `src/processors/section_mapper.py`

- [ ] **Step 1：写映射器**

```python
"""
把 collector 输出的 source 字段映射成 4 个板块。
section ∈ {morning, discussion, github, weekend}

morning:    厂商博客 + 政策 + 中文媒体（一手发布、官方信号）
discussion: HN + Reddit + X + 个人 blog（社区/个人讨论）
github:     GitHub Trending（独立板块，verdict 体系保留）
weekend:    HuggingFace + 长文（沉淀型内容）
"""

MORNING_SOURCES = {
    "OpenAI", "Anthropic", "Google DeepMind", "Meta AI",
    "DeepSeek", "智谱", "Kimi", "通义",
    "网信办", "EU AI Act", "White House AI EO",
    "量子位", "机器之心",
}

DISCUSSION_SOURCES = {
    "HackerNews", "Reddit", "X",
    "Simon Willison", "Lilian Weng", "Eugene Yan", "Chip Huyen",
}

GITHUB_SOURCES = {"GitHub"}
WEEKEND_SOURCES = {"HuggingFace"}


def map_section(item: dict) -> str:
    """优先用 item._section_hint，其次按 source 查表。"""
    hint = item.get("_section_hint")
    if hint in {"morning", "discussion", "github", "weekend"}:
        return hint
    src = item.get("source", "")
    if src in MORNING_SOURCES:
        return "morning"
    if src in DISCUSSION_SOURCES:
        return "discussion"
    if src in GITHUB_SOURCES:
        return "github"
    if src in WEEKEND_SOURCES:
        return "weekend"
    # 未知源默认进 discussion（更安全的兜底）
    return "discussion"


def assign_sections(items: list) -> list:
    """给每个 item 打 section 字段；返回原列表（in-place 改）。"""
    for item in items:
        item["section"] = map_section(item)
        # 清理临时字段
        if "_section_hint" in item:
            del item["_section_hint"]
    return items


def split_by_section(items: list, limits: dict = None) -> dict:
    """按 section 切分，返回 {morning: [...], discussion: [...], github: [...], weekend: [...]}。
    limits = {morning: 8, discussion: 10}; 不在 limits 的 section 不截断。"""
    limits = limits or {}
    out = {"morning": [], "discussion": [], "github": [], "weekend": []}
    # 按 heat_score 降序后再切
    items_sorted = sorted(items, key=lambda x: x.get("heat_score", 0), reverse=True)
    for item in items_sorted:
        s = item.get("section", "discussion")
        if s in out:
            cap = limits.get(s)
            if cap is None or len(out[s]) < cap:
                out[s].append(item)
    return out
```
